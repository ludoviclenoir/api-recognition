for linux
wget https://storage.googleapis.com/learning-datasets/rps.zip -O /tmp/rps.zip

wget https://storage.googleapis.com/learning-datasets/rps-test-set.zip -O /tmp/rps-test-set.zip

script python - extract zip into folder

import os
import zipfile

local_zip = '/tmp/rps.zip'
zip_ref = zipfile.ZipFile(local_zip, 'r')
zip_ref.extractall('/tmp/')
zip_ref.close()

local_zip = '/tmp/rps-test-set.zip'
zip_ref = zipfile.ZipFile(local_zip, 'r')
zip_ref.extractall('/tmp/')
zip_ref.close()
